% code-style.tex — elegante para IEEE
\usepackage{listings}
\usepackage{xcolor}

% paleta sóbria
\definecolor{codebg}{RGB}{250,250,250}
\definecolor{codeframe}{RGB}{220,220,220}
\definecolor{codenum}{RGB}{160,160,160}
\definecolor{codekw}{RGB}{0,0,128}      % azul escuro
\definecolor{codestring}{RGB}{0,128,0}  % verde
\definecolor{codecomment}{RGB}{120,120,120}

\lstdefinestyle{ieeedefault}{
    backgroundcolor=\color{codebg},
    basicstyle=\ttfamily\footnotesize,
    frame=single,
    rulecolor=\color{codeframe},
    frameround=tttt, % cantos arredondados
    breaklines=true,
    tabsize=2,
    showstringspaces=false,
    numbers=left,
    numberstyle=\tiny\color{codenum},
    numbersep=5pt,
    keywordstyle=\bfseries\color{codekw},
    commentstyle=\itshape\color{codecomment},
    stringstyle=\color{codestring},
}

\lstset{style=ieeedefault}
